% =========================================================
% Small helper commands
% These keep formatting choices separate from the text
% content used throughout the thesis.
% =========================================================

% Highlight a keyword in bold
\newcommand{\keyword}[1]{\textbf{#1}}

% Bold table header text
\newcommand{\tabhead}[1]{\textbf{#1}}

% Inline code, commands, or package names
\newcommand{\code}[1]{\texttt{#1}}

% File names in bold monospace
\newcommand{\file}[1]{\texttt{\bfseries #1}}

% Options, arguments, or flags in italic monospace
\newcommand{\option}[1]{\texttt{\itshape #1}}
